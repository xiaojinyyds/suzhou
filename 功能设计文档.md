```javascript
navigator.geolocation.getCurrentPosition(
  (position) => {
    const userLat = position.coords.latitude;
    const userLng = position.coords.longitude;
    const distance = calculateDistance(userLat, userLng, statueLat, statueLng);
    
    if (distance <= statue.radius) {
      // 打卡成功
      checkInSuccess(statue.id);
    } else {
      // 提示距离
      showDistanceAlert(distance);
    }
  },
  (error) => {
    // 定位失败处理
    handleLocationError(error);
  }
);

// 计算两点间距离（Haversine公式）
function calculateDistance(lat1, lng1, lat2, lng2) {
  const R = 6371e3; // 地球半径（米）
  const φ1 = lat1 * Math.PI / 180;
  const φ2 = lat2 * Math.PI / 180;
  const Δφ = (lat2 - lat1) * Math.PI / 180;
  const Δλ = (lng2 - lng1) * Math.PI / 180;

  const a = Math.sin(Δφ/2) * Math.sin(Δφ/2) +
            Math.cos(φ1) * Math.cos(φ2) *
            Math.sin(Δλ/2) * Math.sin(Δλ/2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

  return R * c; // 返回米
}
```

#### 人脸检测与装饰
```javascript
async function detectFaceAndDecorate() {
  const video = document.getElementById('videoElement');
  const canvas = document.getElementById('overlayCanvas');
  const ctx = canvas.getContext('2d');
  
  // 检测人脸
  const detections = await faceapi
    .detectSingleFace(video, new faceapi.TinyFaceDetectorOptions())
    .withFaceLandmarks();
  
  if (detections) {
    const landmarks = detections.landmarks;
    
    // 清空画布
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // 检查是否在圆圈内
    const faceCenter = {
      x: detections.detection.box.x + detections.detection.box.width / 2,
      y: detections.detection.box.y + detections.detection.box.height / 2
    };
    
    const circleCenter = { x: canvas.width / 2, y: canvas.height / 2 };
    const distance = Math.hypot(
      faceCenter.x - circleCenter.x,
      faceCenter.y - circleCenter.y
    );
    
    if (distance < 100) { // 在圆圈内
      // 应用赛博装饰
      drawCyberDecorations(ctx, landmarks);
      updateGuideCircle('success');
    } else {
      updateGuideCircle('default');
    }
  }
  
  // 继续检测（约10fps）
  setTimeout(() => detectFaceAndDecorate(), 100);
}

function drawCyberDecorations(ctx, landmarks) {
  // 1. 绘制光环
  drawHalo(ctx, landmarks.positions[27]); // 鼻子顶部
  
  // 2. 眼睛发光
  drawEyeGlow(ctx, landmarks.getLeftEye());
  drawEyeGlow(ctx, landmarks.getRightEye());
  
  // 3. 电路纹理
  drawCircuitPattern(ctx, landmarks.positions);
  
  // 4. 粒子效果
  drawParticles(ctx, landmarks.positions);
  
  // 5. 扫描线
  drawScanLines(ctx);
}
```

#### 购物车管理
```javascript
class ShoppingCart {
  constructor() {
    this.items = JSON.parse(localStorage.getItem('cart') || '[]');
  }
  
  // 添加商品
  addItem(dessertId, quantity = 1) {
    const existing = this.items.find(item => item.id === dessertId);
    if (existing) {
      existing.quantity += quantity;
    } else {
      this.items.push({
        id: dessertId,
        quantity: quantity,
        addedAt: new Date().toISOString()
      });
    }
    this.save();
    this.updateUI();
  }
  
  // 移除商品
  removeItem(dessertId, quantity = 1) {
    const existing = this.items.find(item => item.id === dessertId);
    if (existing) {
      existing.quantity -= quantity;
      if (existing.quantity <= 0) {
        this.items = this.items.filter(item => item.id !== dessertId);
      }
    }
    this.save();
    this.updateUI();
  }
  
  // 获取总数量
  getTotalQuantity() {
    return this.items.reduce((sum, item) => sum + item.quantity, 0);
  }
  
  // 获取总价格
  getTotalPrice() {
    return this.items.reduce((sum, item) => {
      const dessert = desserts.find(d => d.id === item.id);
      return sum + (dessert.price * item.quantity);
    }, 0);
  }
  
  // 保存到本地存储
  save() {
    localStorage.setItem('cart', JSON.stringify(this.items));
  }
  
  // 更新UI
  updateUI() {
    const badge = document.getElementById('cartBadge');
    const total = this.getTotalQuantity();
    badge.textContent = total;
    badge.style.display = total > 0 ? 'flex' : 'none';
  }
}
```
